#include<iostream>
using namespace std;
struct student{
    int roll;
    char name[20];
    char grade;
    float subject[4];
    };
void display(struct student x)
{
    float sum,avg;
    cout<<"here the details\n";
    cout<<x.name;
    cout<<x.roll;
    cout<<x.grade;
    for(int i=0;i<4;i++)
    {
        cout<<"SUBJECT"<<i+1<<x.subject[i];
        sum+=x.subject[i];
        
    }
    cout<<"Sum is "<<sum;
    avg=sum/4;
    cout<<"average is"<<avg;
}
int main()
{
    struct student x={10,"manisha",'e',{77.22,56.33,56.55,45.22}};
    display(x);
    return 0;
}
